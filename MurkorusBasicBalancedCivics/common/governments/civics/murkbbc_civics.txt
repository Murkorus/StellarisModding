# Civic "Specialisation Incentives"
# Authoritarian / Materialist
# A small bonus to specialist resources from jobs, or slave resources from jobs will be added.
# This civic will allow the empire to adopt a new slavery type, which restricts the workers to only work as specialists, giving slaves a higher amenities upkeep, and increased happiness relative to other slave types.
# A policy option will also be available, allowing the empire to pick between several types of specialist jobs to vastly increase slave resources from jobs in.
# A councillor position will be available which increases resources from specialists for every skill level, and can be employed by officials and scientists.
murkbbc_civic_specialisation_incentives = {
	potential = {
		ethics = {
			NOT = {
				value = ethic_gestalt_consciousness
			}
		}
		authority = {
			NOT = {
				value = auth_corporate
			}
		}
	}
	possible = {
		ethics = {
			OR = {
				text = civic_tooltip_authoritarian
				value = ethic_authoritarian
				value = ethic_fanatic_authoritarian
			}
			OR = {
				text = civic_tooltip_materialist
				value = ethic_materialist
				value = ethic_fanatic_materialist
			}
		}
		civics = {
			NOR = {
				value = civic_pleasure_seekers
				value = civic_free_haven
			}
		}
	}
	random_weight = {
		base = @civic_default_random_weight
	}
	ai_weight = {
		base = @ai_civic_default_base_weight
		modifier = {
			factor = @ai_civic_personality_match_factor
			OR = {
				has_ai_personality = harmonious_hierarchy
				has_ai_personality = decadent_hierarchy
			}
		}
		modifier = {
			factor = @ai_civic_personality_mismatch_factor
			OR = {
				has_ai_personality = erudite_explorers
				has_ai_personality = slaving_despots
			}
		}
	}
	description = "murkbbc_specialisation_incentives_effects"
	modifier = {
		planet_jobs_specialist_produces_mult = 0.05
		add_attunement_the_instrument_of_desire = @civic_attunement1
	}
}
